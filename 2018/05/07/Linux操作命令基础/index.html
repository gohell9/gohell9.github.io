# cd              (change directory  cd )
    cd           home家目录
    cd ~         home家目录
    cd ..          上一级目录
    cd /a/b/c    直接去哪个目录
    cd -        刚才的目录    
    cd /        / 根目录
    cd !$          把上个命令的参数作为cd参数使用。
    cd .config    可进入隐藏文件夹（ls看不到文件夹）



# cp
    cp  f fd                将f复制到fd路径下
    cp  fa fd/fb            将fa文件内容复制到fd路径下fb文件内 
    cp a.txt >> b.txt        把 a.txt追加b.txt后面
    cp a.txt > b.txt        把 a.txt覆盖b.txt 
    cp ../config/* ./



# rm
    rm -rf fa                 把fa文件（文件夹）全部强力删除（-rf  r 递归   f 不提示）
    rm --help                 提示，帮助



# mv 
    mv *1/*2/*3/*4 ../tmp/     将*1/*2/*3/*4几个文件暂时移动到上一级目录的tmp文件夹里面。



# ln
    ln    a.txt a.ln         (创建a.ln为a.txt的硬链接，与源文件同样大小，内容一致。
    ln -s a.txt a.ln     (创建a.ln为a.txt的软链接==符号链接，仅仅指向源文件。

# zip
    sudo apt-get -y install zip
         zip tmp.zip a.txt 将a.txt（可以多个文件）压缩到tmp.zip里面去。（源文件在后面：跟cp，mv等命令源文件位置相反。）
         unzip tmp.zip -d ./tmp     解压到tmp文件夹里

# gzip
    gzip a.txt                         压缩且并不保留源文件，生成名为a.txt.gz的包 
    gzip -c a.txt > a.txt.gz        压缩且并保留源文件，生成名为a.txt.gz的包

    gzip -d a.txt.gz                             解压缩
    gunzip -d a.txt.gz                            解压缩

# tar    （没有压缩功能）  [常用的一般都是 tar的命令!!]
    tar -cvf ./test/sum.tar 1 2 3 4            将 1 2 3 4 打包成sum.tar
    tar -cvzf ./test/sum.tar.gz 1 2 3 4            将 1 2 3 4 压缩打包成sum.tar.gz（常用）    
    tar -xvzf sum.tar.gz                     解压到当前目录
    tar -xvzf sum.tar.gz -C ../tmp             解压到 ../tmp     （常用）
            c     建立新包
            f     命名（f 放在最后）
            v     显示过程信息
            z     压缩，没有没有压缩则tar的参数不需要加z



# -rw-rw-r-- 1 xiaoming xiaoming  136 Dec 31 15:18 sum.tar.gz  (ll举例，不存在这样的)
   -rwxrw-r-- 1
    # - 代表非文件夹 （若是d：代表文件夹）
    # rwx 可读,写，执行 （本用户的权限）
    # rw- 可读,写       （与本用户同组用户的权限）
    # r-- 可读            （其他用户的权限）
        4 2 1  （r=读=4，w=写=2，x=执行=1  ；chmod可更改权限）
    # 1   代表硬链接数（自身也是硬链接）



    # xiaoming             第一个代表权限用户  
    # xiaoming             第二个代表权限用户组员
    # 136                 代表文件大小
    # Dec 31            月份 日期
    # sum.tar.gz        文件名
